rfriends_lxcはxc環境でrfriends3を動作させるスクリプトです  
  
cd ~/  
sudo apt install git  
rm -rf rfriends_lxc  
git clone https://github.com/rfriends/rfriends_lxc.git  
cd rfriends_lxc  
sh rfriends_lxc.sh
